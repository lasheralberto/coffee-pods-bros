"""
Pinecone SDK client singleton for Bounzz platform.
Uses admin.json for config/secrets (with env-var override via Environment).
Supports Pinecone Inference API models (e.g. llama-text-embed-v2)
and OpenAI embedding models.
"""

import json
import logging
import os
from pathlib import Path
from typing import Any, Dict, List, Optional

from pinecone import Pinecone
from pinecone_impl.response_parser import hits_to_json

logger = logging.getLogger(__name__)

_ADMIN_PATH = "pinecone_impl/admin.json"

# Models served by Pinecone Inference (no OpenAI needed)
_PINECONE_INFERENCE_MODELS = {
    "llama-text-embed-v2",
    "multilingual-e5-large",
}


def _load_admin_config() -> Dict[str, Any]:
    """Load the admin.json configuration file."""
    with open(_ADMIN_PATH, "r", encoding="utf-8") as f:
        return json.load(f)


def _resolve(env_var: str, admin_key: str, config: Dict[str, Any], default: str = "") -> str:
    """Return actual os env-var if set, else admin.json value, else *default*."""
    return os.getenv(env_var) or config.get(admin_key, default)


class PineconeClient:
    """Singleton wrapper around Pinecone SDK + embedding generation."""

    _instance: Optional["PineconeClient"] = None

    def __new__(cls) -> "PineconeClient":
        if cls._instance is None:
            cls._instance = super().__new__(cls)
            cls._instance._initialized = False
        return cls._instance

    def __init__(self) -> None:
        if self._initialized:
            return

        self._config = _load_admin_config()

        # Resolve secrets: env vars take priority, then admin.json
        self._api_key: str = _resolve("PINECONE_API_KEY", "pinecone_api_key", self._config)
        self._openai_key: str = _resolve("OPENAI_API_KEY", "openai_api_key", self._config)
        self._index_name: str = _resolve("PINECONE_INDEX_NAME", "pinecone_index_name", self._config, "bounzz")

        self._embedding_model: str = self._config.get("embedding_model", "text-embedding-3-small")
        self._embedding_dim: int = self._config.get("embedding_dimension", 1536)
        self._namespaces: Dict[str, str] = self._config.get("namespaces", {})
        self._use_pinecone_inference: bool = self._embedding_model in _PINECONE_INFERENCE_MODELS

        if not self._api_key:
            logger.warning("PINECONE_API_KEY not set – Pinecone client will not be functional.")
            self._pc = None
            self._index = None
        else:
            self._pc = Pinecone(api_key=self._api_key)
            self._index = self._pc.Index(self._index_name)
            logger.info("Pinecone client initialised (index=%s, model=%s, inference=%s).",
                        self._index_name, self._embedding_model, self._use_pinecone_inference)

        # OpenAI client only needed for non-Pinecone models
        self._openai_client = None
        if not self._use_pinecone_inference:
            if self._openai_key:
                import openai
                self._openai_client = openai.OpenAI(api_key=self._openai_key)
            else:
                logger.warning("OPENAI_API_KEY not set and model is not Pinecone Inference – embeddings will fail.")

        self._initialized = True

    # ------------------------------------------------------------------
    # Namespace helpers
    # ------------------------------------------------------------------
    def resolve_namespace(self, entity_type: str) -> str:
        """Map a logical entity type to a Pinecone namespace."""
        return self._namespaces.get(entity_type, entity_type)

    # ------------------------------------------------------------------
    # Embedding
    # ------------------------------------------------------------------
    def generate_embedding(self, text: str) -> List[float]:
        """Create an embedding vector using Pinecone Inference or OpenAI."""
        if self._use_pinecone_inference:
            if not self._pc:
                raise RuntimeError("Pinecone client not available – check PINECONE_API_KEY.")
            response = self._pc.inference.embed(
                model=self._embedding_model,
                inputs=[text],
                parameters={"input_type": "passage", "truncate": "END"},
            )
            return response.data[0].values
        else:
            if not self._openai_client:
                raise RuntimeError("OpenAI client not available – check OPENAI_API_KEY.")
            response = self._openai_client.embeddings.create(
                input=text,
                model=self._embedding_model,
            )
            return response.data[0].embedding

    # ------------------------------------------------------------------
    # CRUD
    # ------------------------------------------------------------------
    def upsert(
        self,
        entity_type: str,
        entity_id: str,
        text_to_embed: str,
        metadata: Optional[Dict[str, Any]] = None,
    ) -> Dict[str, Any]:
        """
        Generate an embedding for *text_to_embed* and upsert one vector.

        Parameters
        ----------
        entity_type : str
            Logical type (``users``, ``tennisClubs``, ``feed``, …).
        entity_id : str
            Unique identifier (e.g. Firestore doc id).
        text_to_embed : str
            Natural-language text to convert into a vector.
        metadata : dict, optional
            Additional metadata to store alongside the vector.

        Returns
        -------
        dict  with ``upserted_count``.
        """
        if not self._index:
            raise RuntimeError("Pinecone index not available – check PINECONE_API_KEY.")

        namespace = self.resolve_namespace(entity_type)
        vector = self.generate_embedding(text_to_embed)

        meta: Dict[str, Any] = {
            "entity_type": entity_type,
            "entity_id": entity_id,
            "text": text_to_embed[:1000],  # cap stored text
        }
        if metadata:
            meta.update(metadata)

        result = self._index.upsert(
            vectors=[{"id": entity_id, "values": vector, "metadata": meta}],
            namespace=namespace,
        )
        logger.info("Upserted entity_type=%s id=%s ns=%s", entity_type, entity_id, namespace)
        return {"upserted_count": result.get("upserted_count", 1)}

    def upsert_batch(
        self,
        entity_type: str,
        items: List[Dict[str, Any]],
    ) -> Dict[str, Any]:
        """
        Batch upsert. Each item in *items* must have ``id``, ``text``,
        and optionally ``metadata``.
        """
        if not self._index:
            raise RuntimeError("Pinecone index not available – check PINECONE_API_KEY.")

        namespace = self.resolve_namespace(entity_type)
        vectors = []
        for item in items:
            vec = self.generate_embedding(item["text"])
            meta: Dict[str, Any] = {
                "entity_type": entity_type,
                "entity_id": item["id"],
                "text": item["text"][:1000],
            }
            if item.get("metadata"):
                meta.update(item["metadata"])
            vectors.append({"id": item["id"], "values": vec, "metadata": meta})

        result = self._index.upsert(vectors=vectors, namespace=namespace)
        logger.info("Batch upserted %d items in ns=%s", len(vectors), namespace)
        return {"upserted_count": result.get("upserted_count", len(vectors))}

    def delete(self, entity_type: str, entity_id: str) -> Dict[str, Any]:
        """Delete a single vector by id."""
        if not self._index:
            raise RuntimeError("Pinecone index not available – check PINECONE_API_KEY.")

        namespace = self.resolve_namespace(entity_type)
        self._index.delete(ids=[entity_id], namespace=namespace)
        logger.info("Deleted entity_type=%s id=%s ns=%s", entity_type, entity_id, namespace)
        return {"deleted": True}

    def query(
        self,
        entity_type: str,
        text: str,
        top_k: int = 10,
        filter_metadata: Optional[Dict[str, Any]] = None,
    ) -> List[Dict[str, Any]]:
        """Semantic search within a namespace."""
        if not self._index:
            raise RuntimeError("Pinecone index not available – check PINECONE_API_KEY.")

        namespace = self.resolve_namespace(entity_type)
        vector = self.generate_embedding(text)

        results = self._index.search(
            namespace=namespace,
          query={
              "top_k": 40,
              "inputs": {
                  "text": text
              }
          }
        )
        response = hits_to_json(results)
        return response
